202 // The following conditional block of code manually resets as
203 // much of the peripheral set of the LPC43 as possible. This is
204 // done because the LPC43 does not provide a means of triggering
205 // a full system reset under debugger control, which can cause

207 //
208 // You can prevent this code block being included if you require

214  // Disable interrupts
215  __asm volatile ("cpsid i");
216  // equivalent to CMSIS '__disable_irq()' function
217 
218  unsigned int *RESET_CONTROL = (unsigned int *) 0x40053100;
219  // LPC_RGU->RESET_CTRL0 @ 0x40053100
220  // LPC_RGU->RESET_CTRL1 @ 0x40053104
221  // Note that we do not use the CMSIS register access mechanism,
222  // as there is no guarantee that the project has been configured
223  // to use CMSIS.

225  // Write to LPC_RGU->RESET_CTRL0
226  *(RESET_CONTROL + 0) = 0x10DF1000;
227  // GPIO_RST|AES_RST|ETHERNET_RST|SDIO_RST|DMA_RST|
228  // USB1_RST|USB0_RST|LCD_RST|M0_SUB_RST

230  // Write to LPC_RGU->RESET_CTRL1
231  *(RESET_CONTROL + 1) = 0x01DFF7FF;
232  // M0APP_RST|CAN0_RST|CAN1_RST|I2S_RST|SSP1_RST|SSP0_RST|
233  // I2C1_RST|I2C0_RST|UART3_RST|UART1_RST|UART1_RST|UART0_RST|
